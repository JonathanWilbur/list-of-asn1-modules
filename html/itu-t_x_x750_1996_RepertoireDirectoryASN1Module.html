cMISE OBJECT-CLASS ::= {
  KIND          auxiliary
  MUST CONTAIN
    {supportedCmipVersion | supportedCmipProfiles |

sMASE OBJECT-CLASS ::= {
  KIND          auxiliary

  MAY CONTAIN   {mitMoList | supportsMKMglobalNames}
  ID            {mkmDirectoryObjectClass  2}
}

-- The "cMISE" and "sMASE" Directory auxiliary object
